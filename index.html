// --- MOBILE LOGIC --- function renderMobile(items) { const stack = document.getElementById('cardStack'); items.forEach((item, index) => { const card = document.createElement('div'); card.className = 'swipe-card'; card.style.zIndex = items.length - index; card.innerText = item.text; let startX = 0; let currentX = 0; card.addEventListener('touchstart', (e) => { startX = e.touches[0].clientX; card.style.transition = 'none'; }, {passive: true}); card.addEventListener('touchmove', (e) => { currentX = e.touches[0].clientX - startX; const rotation = currentX / 15; // Determine direction: Right = high, Left = low const isDraggingRight = currentX > 0; const isCorrectDir = (isDraggingRight && item.cat === 'high') || (!isDraggingRight && item.cat === 'low'); // 1. Change color based on direction and correctness if (Math.abs(currentX) > 30) { // Threshold to start showing color card.style.backgroundColor = isCorrectDir ? 'var(--correct-green)' : 'var(--error-red)'; card.style.borderColor = isCorrectDir ? 'var(--border-correct)' : 'var(--border-error)'; } else { card.style.backgroundColor = 'white'; card.style.borderColor = 'var(--purple-light)'; } card.style.transform = `translate(-50%, -50%) translateX(${currentX}px) rotate(${rotation}deg)`; }, {passive: true}); card.addEventListener('touchend', () => { const isDraggingRight = currentX > 0; const isCorrectDir = (isDraggingRight && item.cat === 'high') || (!isDraggingRight && item.cat === 'low'); // 2. Prevent swipe-away if the answer is incorrect or movement is too small if (Math.abs(currentX) > 100 && isCorrectDir) { card.style.transition = 'transform 0.3s ease-out, opacity 0.3s'; card.style.transform = `translate(-50%, -50%) translateX(${currentX * 3}px) rotate(${currentX / 2}deg)`; card.style.opacity = '0'; setTimeout(() => card.remove(), 300); handlePlacement(item, isDraggingRight ? 'high' : 'low'); } else { // Snaps back to center if wrong direction or swipe wasn't far enough card.style.transition = 'transform 0.4s cubic-bezier(0.175, 0.885, 0.32, 1.275), background-color 0.3s, border-color 0.3s'; card.style.transform = 'translate(-50%, -50%) translateX(0px) rotate(0deg)'; card.style.backgroundColor = 'white'; card.style.borderColor = 'var(--purple-light)'; } currentX = 0; }); stack.appendChild(card); }); }
